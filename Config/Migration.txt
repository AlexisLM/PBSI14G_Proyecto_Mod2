# Instalar módulos de migrado
drush en migrate_tools -y
drush en migrate_upgrade -y

# Añadir permisos de escritura a settings
chmod +w /sites/default/settings.php

# Agregar conexión al sitio antiguo
Se copia el contenido de la conexión

# Generar migraciones
drush --yes migrate:upgrade --legacy-db-key site1 --legacy-root sites/default/files --configure-only

# Relizar migración
drush migrate-import --all
